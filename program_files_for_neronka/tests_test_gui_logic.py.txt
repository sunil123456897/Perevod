@pytest.fixture
def app():
    with patch('Perevod.project_manager.ProjectManager'), patch('Perevod.database.database_manager.DatabaseManager'), patch('Perevod.knowledge_base.knowledge_base_manager.KnowledgeBaseManager'):
        # Mock the ctk.CTk object to avoid creating a real window
        with patch('customtkinter.CTk') as mock_ctk:
            # Create a mock window instance
            mock_window = MagicMock()
            # Replace the __init__ of TranslatorGUI to not call super().__init__()
            with patch.object(TranslatorGUI, '__init__', lambda x, y=None: None):
                app = TranslatorGUI()
                app.settings_vars = {
                    'project_name': MagicMock(),
                    'api_key': MagicMock(),
                    'embedding_model_name': MagicMock(),
                }
                app.db_manager = None
                app.kb_manager = None
                app.update_progress = MagicMock()
                app.after = MagicMock()
                
                # Mock GUI elements
                app.start_button = MagicMock()
                app.dict_button = MagicMock()
                app.bible_button = MagicMock()
                app.build_index_button = MagicMock()
                
                return app

def test_initialize_thread_default_project(app):
    app.settings_vars['project_name'].get.return_value = "Default"
    app._initialize_thread()
    app.update_progress.assert_called_with(100, "Выберите или создайте проект для начала работы.")
    app.start_button.configure.assert_called_with(state='disabled')

def test_initialize_thread_success(app):
    app.settings_vars['project_name'].get.return_value = "TestProject"
    app.settings_vars['api_key'].get.return_value = "test_api_key"
    app.settings_vars['embedding_model_name'].get.return_value = "test_model"
    
    with patch('Perevod.database.database_manager.DatabaseManager') as mock_db_manager, patch('Perevod.knowledge_base.knowledge_base_manager.KnowledgeBaseManager') as mock_kb_manager:
        app._initialize_thread()
        mock_db_manager.assert_called_once_with("TestProject")
        mock_kb_manager.assert_called_once_with("TestProject", "test_api_key", "test_model")
        app.update_progress.assert_called_with(100, "Инициализация завершена. Готов к переводу.")
        app.start_button.configure.assert_called_with(state='normal')

def test_initialize_thread_exception(app):
    app.settings_vars['project_name'].get.return_value = "TestProject"
    app.settings_vars['api_key'].get.return_value = "test_api_key"
    app.settings_vars['embedding_model_name'].get.return_value = "test_model"
    
    with patch('Perevod.database.database_manager.DatabaseManager', side_effect=Exception("DB Error")):
        app._initialize_thread()
        app.update_progress.assert_called_with(0, "Ошибка инициализации: DB Error")
