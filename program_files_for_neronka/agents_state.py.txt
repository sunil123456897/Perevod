# agents/state.py

from typing import TypedDict, Annotated, List, Dict, Any
from operator import add

# Определяем структуру данных, которая будет передаваться между всеми узлами графа.
# Это единый источник правды для всего рабочего процесса.
class AgentState(TypedDict):
    # Входные данные и конфигурация
    project_name: str
    project_settings: Dict[str, Any]
    progress_callback: Any # Для обновления GUI

    # Очередь задач и результаты
    chapters_to_process: List[Dict[str, str]]
    processed_chapters: Annotated[list, add]
    
    # Служебные поля
    error: str | None
    
    # Менеджеры данных
    db_manager: Any 
    kb_manager: Any
    
    # --- Поля для Конвейера Контроля Качества ---
    # Отчет от агента-аудитора
    quality_assurance_report: Dict[str, Any] | None
    
    # [ИЗМЕНЕНО] Список вердиктов от "Коллегии Судей" для пакетного исправления
    unification_verdicts: Annotated[list, add]
